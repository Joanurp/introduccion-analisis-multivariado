1 Estadístico T2 Hoteling

1.1 Estadístico T2 Hoteling para muestras independientes

consumo <- data.frame(x1 = c(72, 66.5, 54, 67, 44, 41, 34.5, 34.5, 24),
                             x2 = c(50, 48, 57, 60, 57, 52, 50,46, 54), 
                             x3 = c(8, 15, 14, 15, 14, 18, 4, 8.5, 3),
                             x4 = c(0.5, 1.0, 1.0, 0.9, 0.3, 19, 0.5, 1.0, 1.2))

produccion <- data.frame(y1 = c(57, 100, 100, 96.5, 79, 78.5, 48, 155, 84, 105),
                             y2 = c(57, 54, 32, 65, 51, 53, 50, 44, 64, 35), 
                             y3 = c(12.5, 17, 16.5, 20.5, 18, 18, 21, 20.5, 13, 17),
                             y4 = c(0.9, 0.5, 0.7, 0.9, 0.9,1.2, 1.6, 1.4, 0.8, 1.8))


consumo = as.matrix(consumo)
produccion = as.matrix(produccion)

n1=9
n2=10

s1=cov(consumo)
s2=cov(produccion)

spl=((n1-1)*s1+(n2-1)*s2)/(n1+n2-2)

y1=apply(consumo,2,mean)
y2=apply(produccion,2,mean)

t2=((n1*n2)/(n1+n2))*(y1-y2)%*%solve(spl)%*%(y1-y2)

p=4
cal=((n1+n2-p-1)/(p*(n1+n2-2))*t2);cal
         [,1]
[1,] 4.561045
valorp= (pf(cal,p,n1+n2-p-1, lower.tail=F));valorp
           [,1]
[1,] 0.01443042

1.2 Estadístico T2 Hoteling para muestras relacionadas

esmalte1 <- data.frame(x1 = c(73, 43, 47, 53, 58, 47, 52, 38, 61, 56, 56, 34, 55, 65, 75),
                       x2 = c(31, 19, 22, 26, 36, 30, 29, 36, 34, 33, 19, 19, 26, 15, 18))

esmalte2 <- data.frame(y1 = c(51, 41, 43, 41, 47, 32, 24, 43, 53, 52, 57, 44, 57, 40, 68),
                      y2 = c(35, 14, 19, 29, 34, 26, 19, 37, 24, 27, 14, 19, 30, 7, 13))


esmalte1 = as.matrix(esmalte1)
esmalte2 = as.matrix(esmalte2)

dif=esmalte1-esmalte2
dbarra=apply(dif,2,mean)
sd=cov(dif)
n=14

t2=(n*dbarra%*%solve(sd)%*%dbarra);t2
         [,1]
[1,] 10.09764
v=n-1
p=4

cal=((v-p+1)/(v*p))*t2

valorp = (pf(cal,p,v-p+1, lower.tail=F));valorp
          [,1]
[1,] 0.1799308
